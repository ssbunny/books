  // 淘宝刚出这个效果的时候，大家都觉得很惊艳~
  $('#nav').wrap('<div style="position:relative;" id="scrollwrap">');
  $('#scrollwrap').height($('#nav').height());
  $(window).scroll(function(){
    var dom = $('#nav');
      if( $(window).scrollTop() >= $('#scrollwrap').offset().top ){
        dom.css({
          'position': 'fixed',
          'top': '0',
          'z-index': '90000'
        });
      }else{
        dom.css({
        'position': '',
        'top': '',
        'z-index': ''
      });
    }
  });
